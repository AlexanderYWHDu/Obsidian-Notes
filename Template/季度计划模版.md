<%*
let title = tp.file.title;
// 匹配 2026-Q01 这种格式
let match = title.match(/^(\d{4})-Q(\d{2})$/);
let year, quarter;
year = match[1];
quarter = parseInt(match[2], 10);
// 计算该季度起始月份
let startMonth = (quarter - 1) * 3 + 1;
// 生成三个月的文件名
let month = [0, 1, 2].map(i => {
  let m = String(startMonth + i).padStart(2, "0");
  return `${year}-M${m}`;
});
-%>
